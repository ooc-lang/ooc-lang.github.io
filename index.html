ooc is a small programming language with a clear and concise syntax that compiles to C99.

parse: func {
    PropReader new(path, props)

    // parse deps
    requires := props get("Requires")
    if (requires) {
        deps addAll(requires split(',', false) map (|dep| dep trim(" \t")))
    }
}
